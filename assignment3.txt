void loop(int a)
{
int i=1,res=0;

printf("While loop\n");
while(i<=3)
{
res=d*i;
printf("%d * %d = %d\n",d,i,res);
i++;
}
}

void loop1(int a)
{
int res=0,i=1;
int c=a;
printf("Do While loop\n");
do
{
res=c*i;
printf("%d * %d = %d\n",c,i,res);
i++;
}while(i<=3);
}

void loop2(int a)
{
int res=0,i;
